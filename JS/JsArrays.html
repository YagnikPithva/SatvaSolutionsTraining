JavaScript Arrays and Methods

1. Array Creation

            let fruits = ["Apple", "Banana", "Cherry"];

2. Array Properties

            let length = fruits.length; // Get the length of the array

3. push() and pop()

            fruits.push("Orange"); // Adds an element to the end
            let last = fruits.pop(); // Removes the last element

4. shift() and unshift()

            let first = fruits.shift(); // Removes the first element
            fruits.unshift("Mango"); // Adds an element to the beginning

5. concat()

            let tropicalFruits = ["Pineapple", "Papaya"];
            let allFruits = fruits.concat(tropicalFruits); // Merges two arrays

6. join()

            let fruitList = fruits.join(", "); // Joins array elements into a string

7. slice()

            let newArray = fruits.slice(1, 3); // Returns elements from index 1 to 2

8. splice()

            fruits.splice(2, 1, "Grapes"); // Removes 1 item at index 2 and adds "Grapes"

9. forEach()

            fruits.forEach(function(fruit) {
                console.log(fruit); // Prints each fruit in the array
            });

10. map()

            let upperFruits = fruits.map(fruit => fruit.toUpperCase()); // Transforms each fruit to uppercase

11. filter()

            let longNames = fruits.filter(fruit => fruit.length > 5); // Filters fruits with name length > 5

12. reduce()

            let totalLength = fruits.reduce((total, fruit) => total + fruit.length, 0); // Sum of the lengths of all fruit names

13. sort() and reverse()

            fruits.sort(); // Sorts the array alphabetically
            fruits.reverse(); // Reverses the order of elements

14. find() and findIndex()

            let foundFruit = fruits.find(fruit => fruit.startsWith("G")); // Finds the first fruit starting with "G"
            let index = fruits.findIndex(fruit => fruit === "Banana"); // Finds the index of "Banana"

15. includes() and indexOf()

            let hasApple = fruits.includes("Apple"); // Checks if "Apple" is in the array
            let appleIndex = fruits.indexOf("Apple"); // Finds the index of "Apple"
