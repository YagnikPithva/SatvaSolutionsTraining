JavaScript Objects Demonstration

1. Creating an Object

            let person = {
                name: "John",
                age: 30,
                city: "New York"
            };

2. Accessing Object Properties

            let name = person.name;
            let age = person["age"];

3. Adding and Modifying Object Properties

            person.job = "Developer";
            person.age = 35;  // Modifying existing property

4. Methods in Objects

            person.greet = function() {
                return "Hello, my name is " + this.name;
            };

5. The `this` Keyword in Objects

            person.describe = function() {
                return this.name + " is " + this.age + " years old and lives in " + this.city;
            };

6. Object Constructor

            function Person(name, age, city) {
                this.name = name;
                this.age = age;
                this.city = city;
                this.greet = function() {
                    return "Hello, my name is " + this.name;
                };
            }
            let newPerson = new Person("Alice", 28, "Los Angeles");
